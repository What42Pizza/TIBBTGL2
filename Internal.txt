function input
	
	Message = $Args[1]
	OutputVar = $Args[2]
	Min = $Args[3]
	Max = $Args[4]
	
	// enable input box
	$InputBoxText = ""
	$InputBoxReady = false
	$ShowInputBox = true
	
	// take input
	loop
		
		// wait for user
		loop
			PAUSE
			break if $InputBoxReady
		
		// repeat if out of bounds
		InputAsNum = ToNumber $InputBoxText
		if Min and (InputAsNum < Min or InputAsNum > Max) then
			p "Please enter " .. Min .. " though " .. Max .. "."
			repeat
		
		// continue if input is acceptable
		break
	
	// check if input can be converted to a num
	Output = $InputBoxText
	OutputAsNum = ToNumber Output
	if OutputAsNum ~= null then
		Output = OutputAsNum
	
	// set given DataValue
	set OutputVar Output





function print
	do p

function p
	ToPrint = $Args[1]
	AddItem $Log, ToPrint

	



function AddItem
	Table = $Args[1]
	Item = $Args[2]
	Pos = $Args[3]
	NewLength = LengthOf (Table + 1)
	
	if Pos then
		loop i, NewLength, Pos + 1, -1
			Table[i] = Table[i-1]
		Table[Pos] = Item
		return
	
	Table[NewLength] = Item





function exit
	$Continue = false
	PAUSE
